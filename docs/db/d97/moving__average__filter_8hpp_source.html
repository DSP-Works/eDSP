107  void moving_average<T, Allocator>::filter(InputIt first, InputIt last, OutputIt d_first) {
108  if (window_.full()) {
109  for (; first != last; ++d_first, ++first) {
110  accumulated_ -= window_.front();
111  accumulated_ += *first;
112  window_.push_back(*first);
113  *d_first = accumulated_ / static_cast<T>(window_.size());
114  }
115  } else {
116  for (; first != last; ++d_first, ++first) {
117  accumulated_ += *first;

119  *d_first = accumulated_ / static_cast<T>(window_.size());
120  }
121  };
