51  typename Allocator = std::allocator<std::complex<meta::value_type_t<OutputIt>>>>
52  inline void periodogram(InputIt first, InputIt last, OutputIt d_first, SpectralScale scale) {
53  meta::expects(std::distance(first, last) > 0, "Not expecting empty input");
54  using value_type = meta::value_type_t<InputIt>;
55  fftw_plan<value_type> fft_{};
56  const auto size = std::distance(first, last);
57  std::vector<std::complex<value_type>, Allocator> fft_data_(make_fft_size(size), std::complex<value_type>(0, 0));
58  fft_.dft(fftw_cast(&(*first)), fftw_cast(meta::data(fft_data_)), size);
59  if (scale == SpectralScale::Linear) {
60  std::transform(std::cbegin(fft_data_), std::cend(fft_data_), d_first,
61  [](const std::complex<value_type>& val) -> meta::value_type_t<OutputIt> {
62  return math::square(std::abs(val));
63  });
64  } else {
65  std::transform(std::cbegin(fft_data_), std::cend(fft_data_), d_first,
66  [](const std::complex<value_type>& val) -> meta::value_type_t<OutputIt> {
67  return utility::mag2db(std::abs(val));
68  });
69  }
